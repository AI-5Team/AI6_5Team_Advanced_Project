### 로그인

잠금 확인 → 비밀번호 검증 → 실패 카운트 누적 → 성공 시 PBKDF2 → argon2id 자동 rehash.

```python
# services/api/app/services/runtime.py
def login_user(email, password, ip, user_agent):
    row = connection.execute(
        "SELECT * FROM users WHERE email = ? AND deleted_at IS NULL", (email,)
    ).fetchone()

    if row["locked_until"] and row["locked_until"] > utc_now():
        raise api_error(423, "ACCOUNT_LOCKED", "...")   # 잠금 중

    elif not verify_password(password, row["password_hash"]):
        failed_count = (row["failed_login_count"] or 0) + 1
        locked_until = (datetime.now(UTC) + timedelta(minutes=15)).isoformat() \
                       if failed_count >= 5 else None
        connection.execute(
            "UPDATE users SET failed_login_count = ?, locked_until = ? ...",
            (failed_count, locked_until, ...),
        )
        raise api_error(401, "INVALID_CREDENTIALS", "...")

    else:
        # PBKDF2 계정을 argon2id로 자동 rehash
        if not row["password_hash"].startswith("$argon2"):
            new_hash = hash_password(password)
            connection.execute("UPDATE users SET password_hash = ? ...", (new_hash, ...))

    session_token = issue_session(user_id=row["id"], ip=ip, user_agent=user_agent)
    return {"sessionToken": session_token, "user": {...}}
```

### 세션 발급 / 검증

토큰을 SHA-256 해시 후 DB 저장. 쿠키에는 평문만 전달.

```python
# services/api/app/services/session.py
def issue_session(*, user_id, ip, user_agent) -> str:
    token = generate_session_token()          # secrets.token_urlsafe(32)
    token_hash = hash_token(token)            # SHA-256
    expires_at = (datetime.now(UTC) + timedelta(seconds=settings.session_ttl_sec)).isoformat()
    conn.execute(
        "INSERT INTO sessions (id, user_id, session_token_hash, ...) VALUES (...)",
        (str(uuid4()), user_id, token_hash, ip, user_agent, expires_at, ...),
    )
    return token  # 평문 — HttpOnly 쿠키로만 전달

def verify_session(token: str) -> dict | None:
    token_hash = hash_token(token)
    row = conn.execute("""
        SELECT s.*, u.email, u.name, u.status, u.deleted_at
        FROM sessions s JOIN users u ON u.id = s.user_id
        WHERE s.session_token_hash = ?
    """, (token_hash,)).fetchone()
    if not row or row["revoked_at"] or row["expires_at"] < utc_now():
        return None
    return dict(row)
```

### argon2id 해싱

```python
# services/api/app/core/security.py
def hash_password(password: str) -> str:
    if _ARGON2_AVAILABLE:
        return _ph.hash(password)          # argon2id, mem=19456, t=2, p=1
    # fallback: PBKDF2
    salt = secrets.token_hex(16)
    return f"{salt}${_hash_pbkdf2(password, salt)}"

def verify_password(password: str, password_hash: str) -> bool:
    if password_hash.startswith("$argon2"):
        return _ph.verify(password_hash, password)
    # legacy PBKDF2: "salt$digest"
    salt, expected = password_hash.split("$", 1)
    return hmac.compare_digest(expected, _hash_pbkdf2(password, salt))
```

### CSRF 방어 (Origin 검증)

POST/PATCH/DELETE/PUT 요청에 대해 Origin 또는 Referer 헤더를 검증.

```python
# services/api/app/main.py
_CSRF_SAFE_METHODS = {"GET", "HEAD", "OPTIONS"}
_CSRF_EXEMPT_PREFIXES = ("/api/webhooks/",)

@app.middleware("http")
async def csrf_origin_check(request: Request, call_next):
    if request.method not in _CSRF_SAFE_METHODS:
        path = request.url.path
        if not any(path.startswith(p) for p in _CSRF_EXEMPT_PREFIXES):
            origin = request.headers.get("origin") or request.headers.get("referer")
            if origin:
                parsed = urlparse(origin)
                allowed_host = request.url.hostname or ""
                if parsed.hostname and parsed.hostname not in (allowed_host, "localhost", "127.0.0.1"):
                    return JSONResponse(status_code=403, content={"error": {"code": "CSRF_REJECTED", ...}})
    return await call_next(request)
```

### AES-256-GCM 토큰 암호화

OAuth 액세스 토큰을 DB에 저장할 때 암호화. 호출마다 랜덤 IV 생성.

```python
# services/api/app/services/crypto.py
def encrypt_token(plain: str) -> str:
    iv = os.urandom(12)                         # 호출마다 새 IV
    aesgcm = AESGCM(_get_key())                 # AES-256-GCM
    blob = iv + aesgcm.encrypt(iv, plain.encode(), None)
    return b64encode(blob).decode("ascii")      # base64(iv + ciphertext+tag)

def decrypt_token(encoded: str) -> str:
    blob = b64decode(encoded)
    iv, ciphertext = blob[:12], blob[12:]
    return AESGCM(_get_key()).decrypt(iv, ciphertext, None).decode()
```

```typescript
// apps/web/src/middleware.ts
export function middleware(request: NextRequest): NextResponse {
  const { pathname } = request.nextUrl;
  const sessionToken = request.cookies.get("session")?.value;

  if (isPublicPath(pathname)) return addSecurityHeaders(NextResponse.next());

  if (!sessionToken) {
    const loginUrl = new URL("/login", request.url);
    loginUrl.searchParams.set("next", pathname);
    return addSecurityHeaders(NextResponse.redirect(loginUrl));
  }
  return addSecurityHeaders(NextResponse.next());
}
```

```python
def _image_warnings(image: Image.Image) -> list[str]:
```

```python
# services/api/app/services/crypto.py
def get_valid_access_token(user_id: str, provider: str) -> str:
    row = conn.execute(
        "SELECT * FROM social_accounts WHERE user_id = ? AND channel = ?", (user_id, provider)
    ).fetchone()

    if not row or row["status"] not in ("connected", "expired"):
        raise NotConnectedError(f"{provider} 계정이 연결되지 않았습니다.")

    access_token = decrypt_token(row["access_token_ref"])

    if _is_expiring_soon(row["token_expires_at"]):     # 5분 이내 만료 예정
        try:
            refreshed = _simulate_token_refresh(provider, decrypt_token(row["refresh_token_ref"]))
            access_token = refreshed["access_token"]
            conn.execute(
                "UPDATE social_accounts SET access_token_ref = ?, status = 'connected' ... WHERE ...",
                (encrypt_token(access_token), ...),
            )
        except Exception:
            conn.execute("UPDATE social_accounts SET status = 'expired' ... WHERE ...", ...)
            raise NotConnectedError(f"{provider} 토큰 갱신에 실패했습니다. 재연동이 필요합니다.")

    return access_token
```

```python
def _is_token_expired(value: str | None) -> bool:
    expires_at = datetime.fromisoformat(value).replace(tzinfo=timezone.utc)
    return expires_at <= datetime.now(timezone.utc)

def _get_social_account(connection, user_id, channel):
    account = connection.execute(
        "SELECT * FROM social_accounts WHERE user_id = ? AND channel = ?", (user_id, channel)

    # 조회 시점에 만료 확인 → connected → expired 자동 전이
```

```typescript
async function requestJson<T>(url: string, options?: RequestInit): Promise<T> {
  const response = await fetch(url, { credentials: "include", ...options });
  if (!response.ok) {
    if (response.status === 401) {
      clearStoredAuthSession();
      window.location.reload();   // 미들웨어가 /login 으로 리다이렉트
    }
    const data = await response.json().catch(() => null);
    throw new Error(data?.error?.message ?? `HTTP ${response.status}`);
  }
  return response.json();
}
```
